    ```typescript
    import { streamText } from 'ai'
    import { ulid } from 'ulid'
    import { StateMachine } from './state-machine.js'
    import { CircuitBreaker } from './circuit-breaker.js'
    import { createRetryHandler, buildRetryInjection } from './retry.js'
    import { checkToolScope } from '../plan-contract/scope-checker.js'
    import { createTraceLogger } from '../session/trace-logger.js'
    import { saveCheckpoint } from '../session/checkpoint.js'
    import { executeTools } from '@treis/tools'
    import type { ToolCall } from '@treis/tools'
    import type {
      AgentRunOptions,
      AgentEvent,
      StepVerdict,
    } from './types.js'
    import { AgentState, EXECUTION_LIMITS } from './types.js'
    import type { ScopeViolation } from '../plan-contract/scope-checker.js'

    /**
     * Run the agent execution loop for a sealed Plan Contract (AGENT-01 through AGENT-08).
     *
     * The loop:
     * 1. PREPARE: Build messages array with contract context, check compaction
     * 2. STREAM: Stream model output via fullStream, accumulate text + tool calls
     * 3. TOOLS: Dispatch tool calls (scope-checked, circuit-breaker guarded)
     * 4. EVALUATE: Determine step verdict (PASS/FAIL/FATAL)
     * 5. NEXT or COMPLETE: Loop or finish
     *
     * Execution limits: 25 steps max, 10 minutes max, token budget warning.
     * Retry: 3 attempts with 1s/2s/4s backoff, then escalation offer.
     * Circuit breaker: 3 identical tool+input calls -> FATAL.
     * Scope: Pre-hook check before every tool dispatch.
     */
    export async function runAgent(options: AgentRunOptions): Promise<void> {
      const {
        contract,
        tools,
        consumer,
        approveEscalation,
        handleViolation,
        workspace,
        sessionId,
        toolContext,
      } = options

      let currentModel = options.model
      const executionId = ulid()
      const sm = new StateMachine()
      const cb = new CircuitBreaker()
      const retryHandler = createRetryHandler()
      const traceLogger = createTraceLogger(workspace.tracesDir, executionId, sessionId)

      // Per-execution tracking
      let step = 0
      let totalTokensUsed = 0
      let retryCount = 0
      const startTime = Date.now()

      // Conversation history (grows across steps)
      const messages: Array<{ role: string; content: unknown }> = [
        {
          role: 'system',
          content: buildSystemContext(contract),
        },
      ]

      // Build AI SDK tool definitions from Tool[]
      const toolDefinitions = buildToolDefinitions(tools)

      // --- Main execution loop ---
      sm.transition(AgentState.PREPARE)

      while (sm.state !== AgentState.COMPLETE &&
             sm.state !== AgentState.VIOLATED &&
             sm.state !== AgentState.FAILED) {

        // Check execution limits (D-17, AGENT-07)
        if (step >= EXECUTION_LIMITS.MAX_STEPS) {
          consumer({ type: 'failed', reason: `Step limit exceeded (${EXECUTION_LIMITS.MAX_STEPS})`, step })
          sm.transition(AgentState.FAILED)
          break
        }
        if (Date.now() - startTime > EXECUTION_LIMITS.MAX_DURATION_MS) {
          consumer({ type: 'failed', reason: `Time limit exceeded (${EXECUTION_LIMITS.MAX_DURATION_MS}ms)`, step })
          sm.transition(AgentState.FAILED)
          break
        }

        step++

        // --- PREPARE: Compaction check (D-13, AGENT-08) ---
        if (sm.state === AgentState.PREPARE || sm.state === AgentState.NEXT) {
          if (sm.state === AgentState.NEXT) {
            sm.transition(AgentState.PREPARE)
          }
          // Compaction fires only at step boundaries before STREAM
          if (totalTokensUsed > EXECUTION_LIMITS.COMPACTION_THRESHOLD) {
            compactMessages(messages)
          }
          sm.transition(AgentState.STREAM)
        }

        // --- STREAM: Model output via fullStream (D-11, AGENT-02) ---
        const toolCalls: Array<{ toolName: string; toolCallId: string; input: unknown }> = []
        let stepTokens = 0
        let stepText = ''
        let streamError: unknown = null

        try {
          const result = streamText({
            model: currentModel,
            messages: messages as any,
            tools: toolDefinitions,
            maxRetries: 0,  // Agent handles retries, not AI SDK
          })

          for await (const part of result.fullStream) {
            switch (part.type) {
              case 'text-delta':
                consumer({ type: 'token', content: part.textDelta, step })
                stepText += part.textDelta
                break
              case 'tool-call':
                toolCalls.push({
                  toolName: part.toolName,
                  toolCallId: part.toolCallId,
                  input: part.args,
                })
                consumer({ type: 'tool-start', toolName: part.toolName, input: part.args, step })
                break
              case 'finish-step':
                stepTokens += (part.usage?.totalTokens ?? 0)
                break
              case 'error':
                streamError = part.error
                break
            }
          }

          totalTokensUsed += stepTokens
        } catch (err) {
          streamError = err
        }

        // Token budget check (D-04, PLAN-08)
        if (totalTokensUsed > contract.tokenBudget) {
          consumer({ type: 'budget-warning', usedTokens: totalTokensUsed, budgetTokens: contract.tokenBudget })
        }

        // Append assistant response to history
        if (stepText) {
          messages.push({ role: 'assistant', content: stepText })
        }

        // Handle stream errors as FAIL verdict
        if (streamError) {
          const verdict = await handleStepFailure(
            sm, retryHandler, retryCount, step,
            'StreamError', String(streamError),
            consumer, messages, approveEscalation,
            currentModel, options.escalationModel,
            (m) => { currentModel = m },
          )
          if (verdict === 'retry') {
            retryCount++
            sm.transition(AgentState.NEXT)
            continue
          }
          break // FAILED state
        }

        // --- TOOLS: Dispatch with scope checking (D-18, PLAN-06) ---
        if (toolCalls.length > 0) {
          sm.transition(AgentState.TOOLS)

          let violationOccurred = false
          const toolResults: Array<{ toolName: string; success: boolean; output: unknown }> = []

          for (const tc of toolCalls) {
            // Scope pre-hook (D-18, PLAN-06)
            const violation = await checkToolScope(tc.toolName, tc.input, contract)
            if (violation) {
              consumer({ type: 'violation', violation })
              sm.transition(AgentState.VIOLATED)
              violationOccurred = true

              if (handleViolation) {
                const decision = await handleViolation(violation)
                if (decision === 'continue') {
                  // Override for this call only — proceed with tool execution
                  sm.reset()
                  sm.transition(AgentState.PREPARE)
                  sm.transition(AgentState.STREAM)
                  sm.transition(AgentState.TOOLS)
                } else {
                  // 'stop' or 'amend' — halt execution
                  break
                }
              } else {
                break
              }
            }

            if (violationOccurred && sm.state === AgentState.VIOLATED) break

            // Circuit breaker check (D-16, AGENT-06)
            const cbResult = cb.record(tc.toolName, tc.input)
            if (cbResult.triggered) {
              consumer({ type: 'failed', reason: `Circuit breaker: tool "${tc.toolName}" called ${cbResult.count} times with identical input`, step })
              sm.transition(AgentState.EVALUATE)
              sm.transition(AgentState.FAILED)
              violationOccurred = true
              break
            }

            // Dispatch tool via executeTools (D-12, AGENT-03)
            const tool = tools.find(t => t.name === tc.toolName)
            if (!tool) {
              toolResults.push({ toolName: tc.toolName, success: false, output: `Unknown tool: ${tc.toolName}` })
              continue
            }

            const [execResult] = await executeTools([{ tool, input: tc.input }], toolContext)
            const result = execResult!.result

            consumer({
              type: 'tool-result',
              toolName: tc.toolName,
              output: result.data ?? result.error,
              success: result.success,
              step,
            })

            // Trace logging
            traceLogger.logToolCall({
              step,
              tool: tc.toolName,
              input: JSON.stringify(tc.input).slice(0, 200),
              output: JSON.stringify(result.data ?? result.error).slice(0, 500),
              verdict: result.success ? 'PASS' : 'FAIL',
              durationMs: result.durationMs,
            })

            toolResults.push({
              toolName: tc.toolName,
              success: result.success,
              output: result.data ?? result.error,
            })

            // Append tool result to messages for next model call
            messages.push({
              role: 'tool',
              content: JSON.stringify({ toolCallId: tc.toolCallId, result: result.data ?? result.error }),
            })
          }

          if (sm.state === AgentState.VIOLATED || sm.state === AgentState.FAILED) break

          // Evaluate step
          sm.transition(AgentState.EVALUATE)

          const allToolsSucceeded = toolResults.every(r => r.success)
          if (allToolsSucceeded) {
            retryCount = 0
            cb.clear()  // Clear circuit breaker after successful step
            consumer({ type: 'step-complete', step, verdict: 'PASS' })

            // Save checkpoint after successful EVALUATE (SESS-05)
            await saveCheckpoint(workspace.root, {
              stepNumber: step,
              stepStatus: 'completed',
              timestamp: Date.now(),
              executionId,
              state: { totalTokensUsed, messagesLength: messages.length },
            })

            // Check if this was the final step (heuristic: model said end_turn)
            // For Phase 0, we check if the model's finish reason indicates completion
            // The simplest approach: if model produced text with no tool calls, it's done
            sm.transition(AgentState.NEXT)
          } else {
            // Handle step failure
            const verdict = await handleStepFailure(
              sm, retryHandler, retryCount, step,
              'ToolExecutionError',
              toolResults.find(r => !r.success)?.output as string ?? 'Unknown tool failure',
              consumer, messages, approveEscalation,
              currentModel, options.escalationModel,
              (m) => { currentModel = m },
            )
            if (verdict === 'retry') {
              retryCount++
              continue
            }
            break // FAILED state
          }
        } else {
          // Text-only step — no tool calls (Pitfall 5: STREAM -> EVALUATE directly)
          sm.transition(AgentState.EVALUATE)
          retryCount = 0
          consumer({ type: 'step-complete', step, verdict: 'PASS' })

          await saveCheckpoint(workspace.root, {
            stepNumber: step,
            stepStatus: 'completed',
            timestamp: Date.now(),
            executionId,
            state: { totalTokensUsed, messagesLength: messages.length },
          })

          // Text-only response typically means the model is done
          sm.transition(AgentState.COMPLETE)
        }
      }

      // Final event
      if (sm.state === AgentState.COMPLETE) {
        consumer({ type: 'complete', totalSteps: step })
      }

      traceLogger.flush()
    }

    // --- Helper functions ---

    function buildSystemContext(contract: import('../plan-contract/schema.js').PlanContract): string {
      return [
        'You are executing a sealed Plan Contract.',
        `Intent: ${contract.intent}`,
        `Success criteria:`,
        ...contract.successCriteria.map((c, i) => `  ${i + 1}. ${c}`),
        '',
        'Execute the plan step by step. Use the available tools to accomplish each step.',
        'When all success criteria are met, provide a final summary.',
      ].join('\n')
    }

    function buildToolDefinitions(tools: import('@treis/tools').Tool[]): Record<string, any> {
      const defs: Record<string, any> = {}
      for (const tool of tools) {
        defs[tool.name] = {
          description: tool.description,
          parameters: tool.inputSchema,
        }
      }
      return defs
    }

    function compactMessages(messages: Array<{ role: string; content: unknown }>): void {
      // Simple compaction: keep system message + last N messages
      // More sophisticated compaction (LLM-based summary) deferred to Phase 2+
      if (messages.length <= 4) return
      const system = messages[0]! // Keep system prompt
      const recent = messages.slice(-3) // Keep last 3 messages
      messages.length = 0
      messages.push(system, ...recent)
    }

    async function handleStepFailure(
      sm: StateMachine,
      retryHandler: RetryHandler,
      retryCount: number,
      step: number,
      errorType: string,
      errorMessage: string,
      consumer: import('./types.js').AgentConsumer,
      messages: Array<{ role: string; content: unknown }>,
      approveEscalation: ((reason: string) => Promise<boolean>) | undefined,
      currentModel: unknown,
      escalationModel: unknown | undefined,
      setModel: (model: any) => void,
    ): Promise<'retry' | 'failed'> {
      if (retryHandler.shouldRetry(retryCount + 1)) {
        // Retry with error context injection (D-14, AGENT-04)
        const injection = buildRetryInjection('step', errorType, errorMessage, retryCount + 1)
        messages.push(injection)
        consumer({ type: 'retry', step, attempt: retryCount + 1, reason: errorMessage })

        // Backoff delay
        const backoffMs = retryHandler.getBackoffMs(retryCount + 1)
        await new Promise(resolve => setTimeout(resolve, backoffMs))

        // Reset state machine for retry
        if (sm.state === AgentState.EVALUATE) {
          sm.transition(AgentState.NEXT)
        }
        return 'retry'
      }

      // Escalation on 3rd failure (D-15, AGENT-05)
      if (retryHandler.shouldEscalate(retryCount + 1)) {
        const reason = `Step ${step} failed after ${EXECUTION_LIMITS.MAX_RETRIES} retries: ${errorMessage}`
        consumer({ type: 'escalation-required', reason })

        if (approveEscalation && escalationModel) {
          const approved = await approveEscalation(reason)
          if (approved) {
            setModel(escalationModel)
            consumer({ type: 'retry', step, attempt: retryCount + 1, reason: 'Escalated to cloud model' })
            if (sm.state === AgentState.EVALUATE) {
              sm.transition(AgentState.NEXT)
            }
            return 'retry'
          }
        }
      }

      // No more retries, no escalation — FAILED
      consumer({ type: 'failed', reason: errorMessage, step })
      if (sm.state === AgentState.EVALUATE) {
        sm.transition(AgentState.FAILED)
      }
      return 'failed'
    }
    ```
