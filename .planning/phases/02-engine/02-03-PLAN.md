    import { TreisError } from '../errors.js'
    import { AgentState } from './types.js'

    export const TRANSITIONS: Record<AgentState, readonly AgentState[]> = {
      [AgentState.IDLE]:      [AgentState.PREPARE],
      [AgentState.PREPARE]:   [AgentState.STREAM],
      [AgentState.STREAM]:    [AgentState.TOOLS, AgentState.EVALUATE],
      [AgentState.TOOLS]:     [AgentState.EVALUATE],
      [AgentState.EVALUATE]:  [AgentState.NEXT, AgentState.COMPLETE, AgentState.VIOLATED, AgentState.FAILED],
      [AgentState.NEXT]:      [AgentState.PREPARE],
      [AgentState.COMPLETE]:  [],
      [AgentState.VIOLATED]:  [],
      [AgentState.FAILED]:    [],
    }

    export class StateMachine {
      private current: AgentState = AgentState.IDLE

      get state(): AgentState {
        return this.current
      }

      transition(next: AgentState): void {
        const allowed = TRANSITIONS[this.current]
        if (!allowed.includes(next)) {
          throw new TreisError(
            `Illegal state transition: ${this.current} -> ${next}`,
            { timestamp: Date.now() }
          )
        }
        this.current = next
      }

      reset(): void {
        this.current = AgentState.IDLE
      }
    }
    ```

    import { EXECUTION_LIMITS } from './types.js'

    export interface CircuitBreakerResult {
      triggered: boolean
      count: number
      key?: string
    }

    /**
     * Tracks consecutive identical tool calls per execution (D-16, AGENT-06).
     *
     * Key: `${toolName}:${JSON.stringify(input)}`
     * Threshold: 3 consecutive identical calls -> FATAL
     * Cleared after each successful step or on explicit clear().
     */
    export class CircuitBreaker {
      private counts = new Map<string, number>()
      private readonly threshold: number

      constructor(threshold = EXECUTION_LIMITS.CIRCUIT_BREAKER_THRESHOLD) {
        this.threshold = threshold
      }

      record(toolName: string, input: unknown): CircuitBreakerResult {
        const key = `${toolName}:${JSON.stringify(input)}`
        const current = (this.counts.get(key) ?? 0) + 1
        this.counts.set(key, current)

        return {
          triggered: current >= this.threshold,
          count: current,
          key,
        }
      }

      clear(): void {
        this.counts.clear()
      }
    }
    ```

    ```typescript
    import { EXECUTION_LIMITS } from './types.js'

    export interface RetryInjection {
      role: 'user'
      content: string
    }

    /**
     * Build error context injection message for retry (D-14, AGENT-04).
     *
     * Injected as a user message before the next streamText call.
     * Error message truncated to 200 chars per D-14.
     */
    export function buildRetryInjection(
      toolName: string,
      errorType: string,
      errorMessage: string,
      retryCount: number,
    ): RetryInjection {
      return {
        role: 'user',
        content: [
          `[Retry ${retryCount}/${EXECUTION_LIMITS.MAX_RETRIES}] The previous step failed.`,
          `Tool: ${toolName}`,
          `Error type: ${errorType}`,
          `Error: ${errorMessage.slice(0, 200)}`,
          `Please try a different approach or fix the issue.`,
        ].join('\n'),
      }
    }

    export interface RetryHandler {
      shouldRetry(attempt: number): boolean
      shouldEscalate(attempt: number): boolean
      getBackoffMs(attempt: number): number
    }

    /**
     * Create a retry handler with exponential backoff (D-14, AGENT-04).
     *
     * Max 3 retries with 1s/2s/4s backoff.
     * After 3rd failure, shouldEscalate returns true (AGENT-05).
     */
    export function createRetryHandler(): RetryHandler {
      return {
        shouldRetry(attempt: number): boolean {
          return attempt < EXECUTION_LIMITS.MAX_RETRIES
        },

        shouldEscalate(attempt: number): boolean {
          return attempt >= EXECUTION_LIMITS.MAX_RETRIES
        },

        getBackoffMs(attempt: number): number {
          const backoffs = EXECUTION_LIMITS.RETRY_BACKOFF_MS
          return backoffs[Math.min(attempt - 1, backoffs.length - 1)] ?? backoffs[backoffs.length - 1]!
        },
      }
    }
    ```
