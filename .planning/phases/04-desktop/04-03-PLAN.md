Complete the screen router with full state flow:

```typescript
import { useState, useCallback } from 'react'
import { IntentInput } from './screens/IntentInput'
import { Dialogue } from './screens/Dialogue'
import { PlanOptions } from './screens/PlanOptions'
import { SealedContract } from './screens/SealedContract'
import { ExecutionStream } from './screens/ExecutionStream'
import { Result } from './screens/Result'
import { ViolationModal } from './components/ViolationModal'

type Screen = 'intent' | 'dialogue' | 'options' | 'contract' | 'execution' | 'result'

export function App() {
  const [screen, setScreen] = useState<Screen>('intent')
  const [intent, setIntent] = useState('')
  const [clarifications, setClarifications] = useState<Array<{ question: string; answer: string }>>([])
  const [selectedOption, setSelectedOption] = useState<unknown>(null)
  const [contract, setContract] = useState<unknown>(null)
  const [totalSteps, setTotalSteps] = useState(0)

  // Violation modal state
  const [violationOpen, setViolationOpen] = useState(false)
  const [currentViolation, setCurrentViolation] = useState<unknown>(null)

  const handleViolation = useCallback((violation: unknown) => {
    setCurrentViolation(violation)
    setViolationOpen(true)
  }, [])

  const handleViolationDecision = useCallback((decision: 'stop' | 'amend' | 'continue') => {
    setViolationOpen(false)
    setCurrentViolation(null)
    if (decision === 'stop') {
      setScreen('result')
    }
    // amend and continue are handled by IPC bridge (worker resumes)
  }, [])

  const resetToStart = useCallback(() => {
    setScreen('intent')
    setIntent('')
    setClarifications([])
    setSelectedOption(null)
    setContract(null)
    setTotalSteps(0)
  }, [])

  // Subscribe to onStatus for contract-sealed event (from worker)
  // The SealedContract screen triggers execution via IPC
  // When contract-sealed arrives via onStatus, store it

  return (
    <div className="h-screen w-screen bg-[#0F1117] text-[#F0F0F5]" style={{ fontFamily: 'Inter, system-ui, sans-serif' }}>
      {screen === 'intent' && (
        <IntentInput onSubmit={(text) => { setIntent(text); setScreen('dialogue') }} />
      )}
      {screen === 'dialogue' && (
        <Dialogue intent={intent} onComplete={(clarifs) => { setClarifications(clarifs); setScreen('options') }} />
      )}
      {screen === 'options' && (
        <PlanOptions
          intent={intent}
          clarifications={clarifications}
          onSelected={(option) => { setSelectedOption(option); setScreen('contract') }}
        />
      )}
      {screen === 'contract' && (
        <SealedContract contract={contract as any} onBegin={() => setScreen('execution')} />
      )}
      {screen === 'execution' && (
        <ExecutionStream
          onComplete={(steps) => { setTotalSteps(steps); setScreen('result') }}
          onViolation={handleViolation}
        />
      )}
      {screen === 'result' && (
        <Result contract={contract as any} totalSteps={totalSteps} onNewTask={resetToStart} />
      )}
      <ViolationModal
        isOpen={violationOpen}
        violation={currentViolation as any}
        onDecision={handleViolationDecision}
      />
    </div>
  )
}
```

The `contract` state is populated when `PlanOptions.onSelected` fires -- the options screen calls `query({ action: 'execute', data: { intent, clarifications, selectedOption } })` and the worker responds with `contract-sealed` event which the `SealedContract` screen subscribes to via `onStatus`. Wire this: `PlanOptions.onSelected` -> set selectedOption + transition to contract screen. SealedContract listens for `contract-sealed` onStatus event to populate contract data, then renders it.

Focus management: on each screen transition, focus the first interactive element using `autoFocus` or `useEffect` + ref.
