import { createOpenAICompatible } from '@ai-sdk/openai'
import type { ModelAdapter, ModelCapabilities } from './types.js'

const OLLAMA_BASE_URL = 'http://localhost:11434/v1'
const OLLAMA_NUM_CTX = 32768  // Non-negotiable — default 2048 kills agent loops

export function createOllamaAdapter(baseUrl: string = OLLAMA_BASE_URL): ModelAdapter {
  const provider = createOpenAICompatible({
    name: 'ollama',
    baseURL: baseUrl,
  })

  return {
    name: 'ollama',
    providerType: 'ollama',
    getModel(modelId: string) {
      // CRITICAL: num_ctx MUST be passed per-call, not at provider creation
      return provider(modelId, { num_ctx: OLLAMA_NUM_CTX })
    },
    async checkCapabilities(modelId: string): Promise<ModelCapabilities> {
      // Hit Ollama's /api/show endpoint (not OpenAI-compat) to get model info
      const showUrl = baseUrl.replace('/v1', '/api/show')
      const resp = await fetch(showUrl, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name: modelId }),
      })
      if (!resp.ok) throw new Error(`Ollama model check failed: ${resp.status}`)
      const info = await resp.json()
      return {
        supportsToolCalling: true, // Ollama models with tool support vary; check template
        contextWindow: OLLAMA_NUM_CTX,
        modelName: modelId,
      }
    },
  }
}
```

    throw new ModelConnectionError(
      'ANTHROPIC_API_KEY environment variable is not set',

  return {
    name: 'anthropic',
    providerType: 'anthropic',
    getModel(modelId: string) {
      return provider(modelId)

    async checkCapabilities(modelId: string): Promise<ModelCapabilities> {
      return {
        supportsToolCalling: true, // All Anthropic models support tool calling
        contextWindow: 200_000,     // Claude models default; specific models may vary
        modelName: modelId,
      }
    },
  }
}

import type { ModelAdapter, HealthCheckResult } from './adapters/types.js'
import { ModelConnectionError } from '@treis/core'

export async function checkModelHealth(
  adapter: ModelAdapter,
  modelId: string
): Promise<HealthCheckResult> {
  try {
    const capabilities = await adapter.checkCapabilities(modelId)
    return {
      connected: true,
      modelName: capabilities.modelName,
      contextWindow: capabilities.contextWindow,
    }
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error)
    return {
      connected: false,
      modelName: modelId,
      contextWindow: 0,
      error: message,
    }
  }
}
```

Create `packages/api-client/src/slot-manager.ts` per MODEL-06:

```typescript
import type { ModelAdapter, ModelSlot, SlotConfig } from './adapters/types.js'
import { createOllamaAdapter } from './adapters/ollama.js'
import { createAnthropicAdapter } from './adapters/anthropic.js'

export interface SlotManagerConfig {
  slotA: SlotConfig  // strongest model
  slotB: SlotConfig  // fastest model
}

export interface SlotManager {
  getAdapter(slot: ModelSlot): ModelAdapter
  getModelId(slot: ModelSlot): string
  getConfig(slot: ModelSlot): SlotConfig
}

export function createSlotManager(config: SlotManagerConfig): SlotManager {
  const adapters = new Map<ModelSlot, { adapter: ModelAdapter; config: SlotConfig }>()

  function buildAdapter(slotConfig: SlotConfig): ModelAdapter {
    switch (slotConfig.provider) {
      case 'ollama': return createOllamaAdapter()
      case 'anthropic': return createAnthropicAdapter()
      default: throw new Error(`Unknown provider: ${slotConfig.provider}`)
    }
  }

  adapters.set('A', { adapter: buildAdapter(config.slotA), config: config.slotA })
  adapters.set('B', { adapter: buildAdapter(config.slotB), config: config.slotB })

  return {
    getAdapter(slot: ModelSlot): ModelAdapter {
      const entry = adapters.get(slot)
      if (!entry) throw new Error(`No adapter for slot ${slot}`)
      return entry.adapter
    },
    getModelId(slot: ModelSlot): string {
      const entry = adapters.get(slot)
      if (!entry) throw new Error(`No model for slot ${slot}`)
      return entry.config.modelId
    },
    getConfig(slot: ModelSlot): SlotConfig {
      const entry = adapters.get(slot)
      if (!entry) throw new Error(`No config for slot ${slot}`)
      return entry.config
    },
  }
}
```
